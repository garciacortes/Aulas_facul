int main() 
{
	int linha,coluna;
	int matriz[3][3];
	for(linha=0; linha<3; linha++)
	{
		for (linha=0; coluna<3;coluna++)
		{
			printf("Digitar os valores da matriz para: linha %d coluna %d: ",linha,coluna);
		}
	}
